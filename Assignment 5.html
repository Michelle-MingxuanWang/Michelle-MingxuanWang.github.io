# extract file
f = zipfile.ZipFile("./insects.zip",'r')
for file in f.namelist():
    f.extract(file,"./")
f.close()

# picture standardization
batch_size = 32
img_height = 180
img_width = 180

# loading training set and testing set
train = image_dataset_from_directory('insects/train/',
                                                    shuffle = False,
                                                    class_names=["beetles","cockroach","dragonflies"],
                                                   image_size=(img_height, img_width),
                                                   batch_size=batch_size)
test = image_dataset_from_directory('insects/test/',
                                                   shuffle = False,
                                                   class_names=["beetles","cockroach","dragonflies"],
                                                   image_size=(img_height, img_width),
                                                   batch_size=batch_size)

# convert pictures to arrays
X_train = np.concatenate([x for x, y in train], axis = 0)
y_train = np.concatenate([y for x, y in train], axis = 0)
X_test = np.concatenate([x for x, y in test], axis = 0)
y_test = np.concatenate([y for x, y in test], axis = 0)
# normalizing the data 
X_train = X_train / 255
X_test = X_test / 255

classes = ["beetles", "cockroach", "dragonflies"]
def plot_sample(X, y, index):
    plt.figure(figsize = (15,2))
    plt.imshow(X[index])
    plt.xlabel(classes[y[index]])

cnn = models.Sequential([
    layers.Conv2D(filters=32, kernel_size=(3, 3), activation='relu', input_shape=(180, 180, 3)),
    layers.MaxPooling2D((2, 2)),
    
    layers.Conv2D(filters=64, kernel_size=(3, 3), activation='relu'),
    layers.MaxPooling2D((2, 2)),
    
    layers.Flatten(),
    layers.Dense(64, activation='relu'),
    layers.Dense(3, activation='softmax')
])
cnn.compile(optimizer='adam',
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])

cnn.fit(X_train, y_train, epochs=10)
